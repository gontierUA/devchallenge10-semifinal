;(function($) {
    'use strict';
 
    var app = this;
 
    var MODULE = 'rotate';
 
    var DEFAULTS = {
        SELECTORS: {
            MODULE: '.' + MODULE
        },
        CLASSES: {
            ACTIVE: 'active',
            HIDDEN: 'hidden'
        },
        TEXT: {
            ENABLE: 'ENABLE ROTATING',
            DISABLE: 'DISABLE ROTATING'
        }
    };
 
    var Rotate = function(options) {
        this.options = $.extend({}, this.defaults, options);
 
        this.elems = {
            $module: $(DEFAULTS.SELECTORS.MODULE)
        };
 
        this.init();
    };
 
    $.extend(Rotate.prototype, {
        defaults: DEFAULTS,
 
        init: function() {
            app.subscribe(app.EVENTS.ROTATE_OFF, this._rotateOff.bind(this));
            app.subscribe(app.EVENTS.ROTATE_ALLOW, this._rotateAllow.bind(this));
 
            app.subscribe(app.EVENTS.ROTATE_DISALLOW, this._rotateDisallow.bind(this));
            app.subscribe(app.EVENTS.CUSTOM_DRAWING_START, this._rotateDisallow.bind(this));
            app.subscribe(app.EVENTS.CLEAR_ALL, this._rotateDisallow.bind(this));
            app.subscribe(app.EVENTS.SET_BASE_PRODUCT, this._rotateDisallow.bind(this));
 
            this.elems.$module.on('click.Rotate', this._toggleRotation.bind(this));
        },
 
        _rotateAllow: function() {
            this.elems.$module.removeClass(this.options.CLASSES.HIDDEN);
        },
 
        _rotateDisallow: function() {
            this.elems.$module.addClass(this.options.CLASSES.HIDDEN);
        },
 
        _toggleRotation: function() {
            if (this.elems.$module.hasClass(this.options.CLASSES.ACTIVE)) {
                this._rotateOff();
 
                app.publish(app.EVENTS.ROTATE_OFF);
            } else {
                this.elems.$module
                    .addClass(this.options.CLASSES.ACTIVE)
                    .text(this.options.TEXT.DISABLE);
 
                app.publish(app.EVENTS.ROTATE_ON);
            }
        },
 
        _rotateOff: function() {
            this.elems.$module.removeClass(this.options.CLASSES.ACTIVE).text(this.options.TEXT.ENABLE);
        }
    });
 
    var RotateModule = new Rotate();
 
    app.Rotate = Rotate;
}).call(window.DCX = window.DCX || {}, jQuery);
